This Python code defines functions to handle date-related calculations. It checks if a given year is a leap year, calculates the number of days in a given month, considering leap years and calculates the total number of days passed in a given year up to a specific date. Finally, the code prompts the user to input a date and choose between two options: either to calculate the number of days in the given month or the number of days passed in the given year up to that date.

Reflection

The code was straightforward unitl I had to write the days_passed function. I did not know how to calculate the number of days passed in a year so I did research using the different resources and was able to complete the code.

Code


def leap_year(y):
  if y % 4 == 0 and y % 100 != 0 or y % 400 == 0:
    return 1
  else:
    return 0
def number_of_days(m, y):
  if m == 2:
    return 28 + leap_year(y)
  elif m in [4, 6, 9, 11]:
    return 30
  else:
    return 31

def days_passed(d, m, y):
  if leap_year(y)==0:
    days_in_month = [0, 31, 28, 31, 30, 31, 30, 31,
                  31, 30, 31, 30, 31]
    total_days = (sum(days_in_month[:m]) - 1) + d
    return total_days
  elif leap_year(y)==1:
    days_in_month = [0, 31, 29, 31, 30, 31, 30, 31,
                  31, 30, 31, 30, 31]
    total_days = (sum(days_in_month[:m]) - 1) + d
    return total_days

                  days passed in the given year.\n\n"))

  print(number_of_days(month, year))
elif option == 2:
  print(days_passed(day, month, year))

# Function to validate month
def validateMonth(month):
  while month < 1 or month > 12:
    month = int(input("Invalid month. Please enter value from 1-12: "))
    return month

# Function to check if a year is a leap year
def isLeapYear(year):
  if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
    return 1
  else:
    return 0

# Function to validate day
def validateDay(month, day, year):
  while True:
    if month in [4, 6, 9, 11]:
      days_in_month = 30
    elif month == 2:
      if isLeapYear(year):
        days_in_month = 29
      else:
         days_in_month = 28
    else:
      days_in_month = 31

    if day < 1 or day > days_in_month:
      day = int(input(f"Invalid day. Please enter value from 1-{days_in_month}: "))
    else:
      break
  return day

cities = [["Miami", "Atlanta", "Dallas", "Los Angeles"],
["New York", "Chicago", "Portland", "Sacramento"]]
def printList(cities): 
  for row in cities: 
    for col in row: 
      print(col, end=" ") 
    print()

printList(cities)

def flipOrder(cities): 
  cities[1].reverse() 
  cities[0].reverse() 
  printList(cities)
